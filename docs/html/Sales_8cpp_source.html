26std::string Sales::findMatchingPlant(bool outside,
27 bool lowLight,
28 bool lowWater,
29 bool brightColour,
30 bool lowCare)
31{
32 Inventory* inv = Inventory::getInstance();
33 std::vector<std::string> matchingPlants;
34
35 // Check each candidate plant
36 for (int i = 0; i < NUM_CANDIDATES; ++i) {
37 std::string plantName = CANDIDATE_NAMES[i];
38 Plant* plant = inv->getPrototype(plantName);
39 if(!plant) continue;
40
41 std::shared_ptr<PlantData> data = plant->getPlantData();
42 if(!data) continue;
43
44 bool matches = true;
45
46 if (outside && !data->isOutside()) matches = false;
47 if (lowLight && !data->isLowLight()) matches = false;
48 if (lowWater && !data->isLowWater()) matches = false;
49 if (brightColour && !data->isBrightColour()) matches = false;
50 if (lowCare && !data->isLowCare()) matches = false;
51
52 if(matches) {
53 int stock = inv->getStock(plantName);
54 if(stock > 0) {
55 matchingPlants.push_back(plantName);
56 }
57 }
58 }
59
60 if(!matchingPlants.empty()) {
61 std::string plantName = matchingPlants[0];
62 int stock = inv->getStock(plantName);
63 return "We have " + plantName + " which matches your needs. We have " + std::to_string(stock) + " in stock.";
64 }
65
66 return "We don't have that plant in stock right now, but we can order it for you!";
67}
