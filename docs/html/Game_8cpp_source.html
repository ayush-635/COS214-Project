426Order* Game::createOrderFromReadyPlants() {
427 Order* order = new Order();
428 int plantsAdded = 0;
429 int maxPlants = 1 + (rand() % 3); // 1-3 plants
430
431 for (auto space : spaces) {
432 if (plantsAdded >= maxPlants) break;
433
434 for (int i = 0; i < 10 && plantsAdded < maxPlants; ++i) {
435 PlantableArea* child = space->getChild(i);
436 if (!child) break;
437
438 PlanterBox* box = dynamic_cast<PlanterBox*>(child);
439 if (box) {
440 auto plants = box->getPlants();
441 for (auto plant : plants) {
442 if (plant->getStateName() == "ReadyToSell" && plantsAdded < maxPlants) {
443 double price = 25.0 + (rand() % 26); // R25-R50
444 OrderItem* item = new OrderItem(plant->getName(), price, 1);
445
446 // Add pot with 50% chance (Decorator pattern)
447 if (rand() % 100 < 50) {
448 item->addPot();
449 }
450
451 order->addItem(item);
452
453 // Remove sold plant
454 box->removePlant(plant);
455 delete plant;
456 plantsAdded++;
457 break;
458 }
459 }
460 }
461 }
462 }
463
464 return order;

721bool Game::waterPlants(int spaceIndex, int boxIndex) {
722 if (spaceIndex < 0 || spaceIndex >= (int)spaces.size()) return false;
723
724 // Determine water needed based on strategy
725 int waterNeeded = 1; // Light
726 if (dynamic_cast<IntermediateWateringStrategy*>(currentWateringStrategy)) {
727 waterNeeded = 3;
728 } else if (dynamic_cast<HeavyWateringStrategy*>(currentWateringStrategy)) {
729 waterNeeded = 5;
730 }
731
732 // Check water availability FIRST
733 if (!resourceManager->useWater(waterNeeded)) {
734 return false;
735 }
736
737 // Check money
738 if (bankAccount->getBalance() < 1) {
739 std::cout << "‚ùå Insufficient funds! Need R1" << std::endl;

743 }

746 if (!collection) return false;
747
748 bankAccount->withdraw(1, "Water");
